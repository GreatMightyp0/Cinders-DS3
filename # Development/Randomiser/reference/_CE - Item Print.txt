{$lua}
for _,v in ipairs(listToShow) do
    local hex = tohex08(v.id)
    local sub_id = string.sub(v.id, -4)
    if v.id >= 1000000 and v.id <= 40000000 then
       if sub_id == '0000' then
          print( hex .. ' ' .. v.name )
       end
    else
        print( hex .. ' ' .. v.name )
    end
end
